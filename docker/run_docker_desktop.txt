---------------------
RUN MINIMA ON DESKTOP
---------------------

#FIRST INSTALL DOCKER DESKTOP
https://docs.docker.com/get-docker/


#THEN - Run this command in terminal for each device
#You MUST set a new password at the bit that says 'minima_mdspassword=123' - change that to whatever you want

#Your files, backups etc will be stored in a folder named 'MinimaDocker' in your HOME directory

# WINDOWS
docker run -d -e minima_mdspassword=123 -e minima_desktop=true -v $HOME/MinimaDocker:/home/minima/data -p 8001-8004:9001-9004 --restart unless-stopped --name minima minimaglobal/minima:latest

# MAC
docker run -d -e minima_mdspassword=123 -e minima_desktop=true -v ~/MinimaDocker:/home/minima/data -p 8001-8004:9001-9004 --restart unless-stopped --name minima minimaglobal/minima:latest

# LINUX
docker run -d -e minima_mdspassword=123 -e minima_desktop=true -v ~/MinimaDocker:/home/minima/data -p 8001-8004:9001-9004 --restart unless-stopped --name minima minimaglobal/minima:latest


#THEN - Run this command in terminal so that you are automatically updated when a new version is released

# WATCHTOWER - update minima docker when new image pushed
docker run -d --restart unless-stopped --name watchtower -e WATCHTOWER_CLEANUP=true -e WATCHTOWER_TIMEOUT=60s -v /var/run/docker.sock:/var/run/docker.sock containrrr/watchtower
